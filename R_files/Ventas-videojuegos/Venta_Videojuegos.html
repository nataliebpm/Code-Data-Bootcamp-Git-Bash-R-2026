Tema visual

theme_videojuegos <- function() {
  theme_minimal(base_size = 13) +
    theme(
      plot.title = element_text(face = "bold", size = 16),
      plot.subtitle = element_text(size = 11),
      plot.caption = element_text(size = 9, color = "#659287"),
      axis.title = element_text(face = "bold"),
      panel.grid.minor = element_blank(),
      legend.position = "bottom"
    )
}

Ejercicio 04: Venta típica por género

ventas_tipicas_genero <- videojuegos_limpio %>%
  group_by(Genre) %>%
  summarise(
    venta_promedio = mean(Global_Sales, na.rm = TRUE),
    venta_mediana = median(Global_Sales, na.rm = TRUE),
    numero_juegos = n(),
    .groups = "drop"
  ) %>%
  arrange(desc(venta_mediana))
